UpRi = list(map(lambda a : int(a), input().split(" ")))
inp = []
xC = 0
yC = 0


while inp != [0, 0]:
    inp = list(map(lambda a : int(a), input().split(" ")))
    
    xC += inp[0]
    yC += inp[1]
    
    if xC < 0:
        xC = 0
        
    if xC > UpRi[0]:
        xC = UpRi[0]
    
    if yC < 0:
        yC = 0
    
    if yC > UpRi[1]:
        yC = UpRi[1]
    
    if inp != [0, 0]:
        print (xC, yC)
